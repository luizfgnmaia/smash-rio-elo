query EventSets($eventId: ID!, $page:Int!, $perPage:Int!){
  event(id:$eventId){
    tournament{
      name
      id
    }
    sets(
      page: $page
      perPage: $perPage
      sortType: MAGIC
    ){
      nodes{
        completedAt
        entrant1Score
        entrant2Score
        slots{
          entrant{
            participants{
              gamerTag
              playerId
            }
          }
        }
      }
    }
  }
},
